# linked-role-lib

## Installation

```sh
# pnpm
pnpm install linked-role-lib

# Bun
bun install linked-role-lib
```

<details>
<summary>npm, Yarn</summary>

```sh
# npm
npm install linked-role-lib

# yarn
yarn add linked-role-lib
```
</details>

export * from 'npm:linked-role-lib';

  }
}
